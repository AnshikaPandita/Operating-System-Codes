#include <stdio.h> 
#include <stdlib.h> 
#include <sys/types.h> 
#include <sys/wait.h> 
#include <unistd.h> 

int main (int argc, char *argv[])
{
int a[3] = {1,2,3};
printf("\nChild process created.\n"); 
 printf("\nChild => PPID:%d PID: %d\n", getppid(), getpid()); 
 for(int i = 0; i<3 ;i++) 
 { 
 a[i] = a[i] * a[i] * a[i]; 
 } 

printf("\nCube of numbers in array:\t"); 
for(int i=0;i<3;++i) 
{ 
printf("%d\t",a[i]); 
} 
printf("\n"); 
exit(0);
return 0;
}
