#include <stdio.h> 
#include <stdlib.h> 
#include <sys/types.h> 
#include <sys/wait.h> 
#include <unistd.h> 

int main (int argc, char *argv[])
{
int a[3] = {1,2,3}; 
pid_t p = fork();

if(p<0)
{
printf("error");
}
if(p>0){
printf("\n We are in Parent Process waiting for child to execute.\n");
//wait(NULL); 
printf("\nParent => PPID: %d PID: %d\n", getppid(), getpid());  

for(int i = 0; i<3 ;i++) 
 { 
 a[i] = a[i] * a[i]; 
 } 

printf("\nSquare of numbers in array:\t");

for(int i=0;i<3;++i) 
{ 
printf("%d\t",a[i]); 
} 
printf("\n"); 
}
if(p == 0){
char *args[]={"./child.c",NULL};

execv("./child",args);
}
return 0;
}
