#include<stdio.h>
#include<dirent.h>
#include<stdlib.h>
#include <sys/stat.h>
#include <time.h>
#include<sys/types.h>

void getFileCreationTime(char*Path){
   struct stat attr;
   stat(Path,&attr);
   printf("last modified time:%s",ctime(&attr.st_mtime));


}
int main(){
char dirname[10];
DIR*p;
struct dirent*d;
printf("Enter the directory name:");
scanf("%s",dirname);
p=opendir(dirname);
if(p==NULL){
printf("Cannot find directory!");
exit(-1);
}
while(d=readdir(p)){

printf("%s\n",d->d_name);
 getFileCreationTime(dirname);
}
return 0;
}
